{{- $page := .page -}} {{- $marginable := .marginable -}} {{- $side := .side | default "right" -}} {{- $source := "" -}} {{- with .body -}} {{- $source = . -}} {{- else -}} {{- $source = .inner -}} {{- end -}} {{- $inner := $source | strings.TrimSpace -}} {{- $notes := $page.Scratch.Get "tufteNotes" | default (slice) -}} {{- $number := add (len $notes) 1 -}} {{- $inlineBody := $inner | $page.RenderString (dict "display" "inline") -}} {{- $blockBody := $inner | $page.RenderString (dict "display" "block") -}} {{- $normalized := replaceRE `\r\n?` "\n" $inner -}} {{- $normalized = replaceRE `\n[ \t]*\n+` "\n\n" $normalized -}} {{- $marginParagraphs := slice -}} {{- range split $normalized "\n\n" -}} {{- $paragraph := . | strings.TrimSpace -}} {{- if $paragraph -}} {{- $marginParagraphs = $marginParagraphs | append ($paragraph | $page.RenderString (dict "display" "inline")) -}} {{- end -}} {{- end -}} {{- $marginLead := $inlineBody -}} {{- $marginTail := slice -}} {{- if gt (len $marginParagraphs) 0 -}} {{- $marginLead = index $marginParagraphs 0 -}} {{- if gt (len $marginParagraphs) 1 -}} {{- $marginTail = after 1 $marginParagraphs -}} {{- end -}} {{- end -}} {{- $backref := printf `↩` $number -}} {{- $endnoteBody := printf `%s %s` $blockBody $backref | safeHTML -}} {{- if gt (len (findRE `

\s*$` $blockBody)) 0 -}} {{- $endnoteBody = replaceRE `

\s*$` (printf ` %s

` $backref) $blockBody | safeHTML -}} {{- end -}} {{- $note := dict "number" $number "marginable" $marginable "side" $side "blockBody" $blockBody "endnoteBody" $endnoteBody -}} {{- $page.Scratch.Add "tufteNotes" (slice $note) -}} {{- $classes := printf "tufte-note%s" (cond $marginable (printf " tufte-note--marginable tufte-note--%s" $side) " tufte-note--endnote") -}} {{- $markup := printf `%d` $classes $number $number $number -}} {{- if $marginable -}} {{- $markup = printf `%s%d.%s` $markup $side $number $marginLead -}} {{- range $marginTail -}} {{- $markup = printf `%s%s` $markup . -}} {{- end -}} {{- $markup = printf `%s` $markup -}} {{- end -}} {{- $markup = printf `%s` $markup -}} {{- $markup | safeHTML -}}
