{{- $ctx := .ctx -}} {{- $tag := .tag -}} {{- $align := lower (strings.TrimSpace (default "" ($ctx.Get "align"))) -}} {{- if and $align (not (in (slice "left" "center" "right") $align)) -}} {{- errorf "%s align must be left, center, or right at %s" $ctx.Name $ctx.Position -}} {{- end -}} {{- $numeric := eq (lower (printf "%v" (default false ($ctx.Get "numeric")))) "true" -}} {{- $nowrap := eq (lower (printf "%v" (default false ($ctx.Get "nowrap")))) "true" -}} {{- $label := eq (lower (printf "%v" (default false ($ctx.Get "label")))) "true" -}} {{- $corner := eq (lower (printf "%v" (default false ($ctx.Get "corner")))) "true" -}} {{- $rowLabel := strings.TrimSpace (default "" ($ctx.Get "rowlabel")) -}} {{- $colLabel := strings.TrimSpace (default "" ($ctx.Get "collabel")) -}} {{- $rowspan := strings.TrimSpace (printf "%v" (default "" ($ctx.Get "rowspan"))) -}} {{- $colspan := strings.TrimSpace (printf "%v" (default "" ($ctx.Get "colspan"))) -}} {{- if and $corner (ne $tag "th") -}} {{- errorf "corner is only valid on tabula-th at %s" $ctx.Position -}} {{- end -}} {{- if and (or $rowLabel $colLabel) (not $corner) -}} {{- errorf "rowlabel and collabel are only valid with corner=true on tabula-th at %s" $ctx.Position -}} {{- end -}} {{- if and $corner (not (or $rowLabel $colLabel)) -}} {{- errorf "tabula-th corner requires rowlabel=, collabel=, or both at %s" $ctx.Position -}} {{- end -}} {{- if and $rowspan (not (findRE `^[1-9]\d*$` $rowspan)) -}} {{- errorf "%s rowspan must be a positive integer at %s" $ctx.Name $ctx.Position -}} {{- end -}} {{- if and $colspan (not (findRE `^[1-9]\d*$` $colspan)) -}} {{- errorf "%s colspan must be a positive integer at %s" $ctx.Name $ctx.Position -}} {{- end -}} {{- $outcome := lower (strings.TrimSpace (default "" ($ctx.Get "outcome"))) -}} {{- if and $outcome (not (in (slice "yes" "no" "partial") $outcome)) -}} {{- errorf "%s outcome must be yes, no, or partial at %s" $ctx.Name $ctx.Position -}} {{- end -}} {{- $scope := "" -}} {{- if eq $tag "th" -}} {{- $scope = lower (strings.TrimSpace (default "" ($ctx.Get "scope"))) -}} {{- if and (not $scope) (not $corner) -}} {{- $scope = "col" -}} {{- end -}} {{- if and $scope (not (in (slice "row" "col" "rowgroup" "colgroup") $scope)) -}} {{- errorf "tabula-th scope must be row, col, rowgroup, or colgroup at %s" $ctx.Position -}} {{- end -}} {{- else if $ctx.Get "scope" -}} {{- errorf "scope is only valid on tabula-th at %s" $ctx.Position -}} {{- end -}} {{- $classes := slice "tabula__cell" -}} {{- with $align -}} {{- $classes = $classes | append (printf "tabula__cell--%s" .) -}} {{- end -}} {{- if $numeric -}} {{- $classes = $classes | append "tabula__cell--numeric" -}} {{- end -}} {{- if $nowrap -}} {{- $classes = $classes | append "tabula__cell--nowrap" -}} {{- end -}} {{- if $corner -}} {{- $classes = $classes | append "tabula__cell--matrix-corner" -}} {{- end -}} {{- if or $label $outcome -}} {{- $classes = $classes | append "tabula__cell--label" -}} {{- end -}} {{- with $outcome -}} {{- $classes = $classes | append (printf "outcome--%s" .) -}} {{- end -}} {{- $inner := .inner | strings.TrimSpace -}} {{- $allowEmpty := and (eq $tag "th") (in (slice "colgroup" "rowgroup") $scope) -}} {{- if and (not $corner) (not $allowEmpty) (not $inner) -}} {{- errorf "%s requires inner content at %s" $ctx.Name $ctx.Position -}} {{- end -}} {{- $body := "" -}} {{- if $corner -}} {{- $cornerClasses := slice "tabula__matrix-corner" -}} {{- $cornerParts := slice -}} {{- if $colLabel -}} {{- $cornerClasses = $cornerClasses | append "tabula__matrix-corner--has-col" -}} {{- $renderedColLabel := $colLabel | $ctx.Page.RenderString (dict "display" "inline") -}} {{- $cornerParts = $cornerParts | append (printf `%s` $renderedColLabel) -}} {{- end -}} {{- if $rowLabel -}} {{- $cornerClasses = $cornerClasses | append "tabula__matrix-corner--has-row" -}} {{- $renderedRowLabel := $rowLabel | $ctx.Page.RenderString (dict "display" "inline") -}} {{- $cornerParts = $cornerParts | append (printf `%s` $renderedRowLabel) -}} {{- end -}} {{- $body = printf `%s` (delimit $cornerClasses " ") (delimit $cornerParts "") | safeHTML -}} {{- else -}} {{- $body = $inner | $ctx.Page.RenderString (dict "display" "inline") -}} {{- end -}} {{- if eq $tag "th" -}} {{ $body }} {{- else -}} {{ $body }} {{- end -}}
