{{- $caption := strings.TrimSpace (default "" (.Get "caption")) -}} {{- $label := strings.TrimSpace (default $caption (.Get "label")) -}} {{- if not $label -}} {{- errorf "tabula requires caption= or label= at %s" .Position -}} {{- end -}} {{- $wide := eq (lower (printf "%v" (default false (.Get "wide")))) "true" -}} {{- $matrix := eq (lower (printf "%v" (default false (.Get "matrix")))) "true" -}} {{- $striped := eq (lower (printf "%v" (default false (.Get "striped")))) "true" -}} {{- $compact := eq (lower (printf "%v" (default false (.Get "compact")))) "true" -}} {{- $ornament := eq (lower (printf "%v" (default false (.Get "ornament")))) "true" -}} {{- $figureClasses := slice "tabula-frame" -}} {{- if $wide -}} {{- $figureClasses = $figureClasses | append "tabula-frame--wide" -}} {{- end -}} {{- $tableClasses := slice "tabula" -}} {{- if $matrix -}} {{- $tableClasses = $tableClasses | append "tabula--matrix" -}} {{- end -}} {{- if $striped -}} {{- $tableClasses = $tableClasses | append "tabula--striped" -}} {{- end -}} {{- if $compact -}} {{- $tableClasses = $tableClasses | append "tabula--compact" -}} {{- end -}} {{- if $ornament -}} {{- $tableClasses = $tableClasses | append "tabula--ornamented" -}} {{- end -}} {{- $style := "" -}} {{- with .Get "minwidth" -}} {{- $minwidth := strings.TrimSpace . -}} {{- if findRE `^\d+(?:\.\d+)?(?:rem|em|ch|px)$` $minwidth -}} {{- $style = printf `--tabula-min-width: %s;` $minwidth -}} {{- else -}} {{- errorf "tabula minwidth must look like 42rem, 60ch, or 720px at %s" $.Position -}} {{- end -}} {{- end -}} {{- $captionID := printf "tabula-caption-%d" .Ordinal -}} {{- $noteID := printf "tabula-note-%d" .Ordinal -}} {{- $note := strings.TrimSpace (default "" (.Get "note")) -}}
