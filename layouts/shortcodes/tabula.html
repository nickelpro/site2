label= at %s" .Position -}} {{- end -}} {{- $wide := eq (lower (printf "%v" (default false (.Get "wide")))) "true" -}} {{- $matrix := eq (lower (printf "%v" (default false (.Get "matrix")))) "true" -}} {{- $striped := eq (lower (printf "%v" (default false (.Get "striped")))) "true" -}} {{- $compact := eq (lower (printf "%v" (default false (.Get "compact")))) "true" -}} {{- $ornament := eq (lower (printf "%v" (default false (.Get "ornament")))) "true" -}} {{- $figureClasses := slice "tabula-frame" -}} {{- if $wide -}} {{- $figureClasses = $figureClasses | append "tabula-frame--wide" -}} {{- end -}} {{- $tableClasses := slice "tabula" -}} {{- if $matrix -}} {{- $tableClasses = $tableClasses | append "tabula--matrix" -}} {{- end -}} {{- if $striped -}} {{- $tableClasses = $tableClasses | append "tabula--striped" -}} {{- end -}} {{- if $compact -}} {{- $tableClasses = $tableClasses | append "tabula--compact" -}} {{- end -}} {{- if $ornament -}} {{- $tableClasses = $tableClasses | append "tabula--ornamented" -}} {{- end -}} {{- $styleParts := slice -}} {{- $tabulaWidthPattern := `^\d+(?:\.\d+)?(?:rem|em|ch|px|%)$` -}} {{- $tabulaAbsoluteWidthPattern := `^\d+(?:\.\d+)?(?:rem|em|ch|px)$` -}} {{- $tabulaPercentWidthPattern := `^\d+(?:\.\d+)?%$` -}} {{- with .Get "minwidth" -}} {{- $minwidth := strings.TrimSpace . -}} {{- if findRE $tabulaWidthPattern $minwidth -}} {{- $styleParts = $styleParts | append (printf `--tabula-min-width: %s;` $minwidth) -}} {{- else -}} {{- errorf "tabula minwidth must look like 80%%, 42rem, 60ch, or 720px at %s" $.Position -}} {{- end -}} {{- end -}} {{- with .Get "maxwidth" -}} {{- $maxwidth := strings.TrimSpace . -}} {{- if findRE $tabulaAbsoluteWidthPattern $maxwidth -}} {{- $styleParts = $styleParts | append (printf `--tabula-max-width: %s;` $maxwidth) -}} {{- else if findRE $tabulaPercentWidthPattern $maxwidth -}} {{- $maxwidthPercent := replaceRE `%$` "" $maxwidth -}} {{- $styleParts = $styleParts | append (printf `--tabula-max-width: calc(var(--content-column-max-width, 38rem) * (%s / 100));` $maxwidthPercent) -}} {{- else -}} {{- errorf "tabula maxwidth must look like 80%%, 42rem, 60ch, or 720px at %s" $.Position -}} {{- end -}} {{- end -}} {{- $style := delimit $styleParts " " -}} {{- $captionID := printf "tabula-caption-%d" .Ordinal -}} {{- $noteID := printf "tabula-note-%d" .Ordinal -}} {{- $note := strings.TrimSpace (default "" (.Get "note")) -}}
